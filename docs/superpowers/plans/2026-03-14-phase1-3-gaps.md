- [ ] **Step 4: Implement expression evaluator**

Create `lib/core/utils/expression_evaluator.dart`:

```dart
/// Evaluates a simple arithmetic expression string with
/// operator precedence (multiply/divide before add/subtract).
///
/// Supports operators: + − (U+2212) × (U+00D7) ÷ (U+00F7)
/// Returns 0.0 for empty or invalid expressions.
double evaluateExpression(String expr) {
  if (expr.isEmpty) return 0.0;

  final tokens = _tokenize(expr);
  if (tokens.isEmpty) return 0.0;

  return _evaluate(tokens);
}

List<String> _tokenize(String expr) {
  final tokens = <String>[];
  final buf = StringBuffer();

  for (final ch in expr.split('')) {
    if (_isOperator(ch)) {
      if (buf.isNotEmpty) {
        tokens.add(buf.toString());
        buf.clear();
      }
      tokens.add(ch);
    } else {
      buf.write(ch);
    }
  }
  if (buf.isNotEmpty) tokens.add(buf.toString());
  return tokens;
}

bool _isOperator(String ch) {
  return ch == '+' || ch == '\u2212' || ch == '\u00D7' || ch == '\u00F7';
}

double _evaluate(List<String> tokens) {
  // Phase 1: handle × and ÷
  final simplified = <String>[];
  var i = 0;
  while (i < tokens.length) {
    if (i + 2 < tokens.length &&
        (tokens[i + 1] == '\u00D7' || tokens[i + 1] == '\u00F7')) {
      var left = double.tryParse(
            simplified.isNotEmpty ? simplified.removeLast() : tokens[i],
          ) ??
          0;
      while (i + 2 < tokens.length &&
          (tokens[i + 1] == '\u00D7' || tokens[i + 1] == '\u00F7')) {
        final op = tokens[i + 1];
        final right = double.tryParse(tokens[i + 2]) ?? 0;
        if (op == '\u00D7') {
          left = left * right;
        } else {
          left = right == 0 ? 0 : left / right;
        }
        i += 2;
      }
      simplified.add(left.toString());
      i++;
    } else {
      simplified.add(tokens[i]);
      i++;
    }
  }

  // Phase 2: handle + and −
  if (simplified.isEmpty) return 0.0;
  var result = double.tryParse(simplified[0]) ?? 0.0;
  var j = 1;
  while (j + 1 < simplified.length) {
    final op = simplified[j];
    final right = double.tryParse(simplified[j + 1]) ?? 0;
    if (op == '+') {
      result += right;
    } else if (op == '\u2212') {
      result -= right;
    }
    j += 2;
  }
  return result;
}
```

  BuildContext context,
